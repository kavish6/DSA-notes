long long getMaxArea(long long arr[], int n)
    {
        // Your code here
        stack<int> st;
        long long max_ar=0;
        int i=0;
        while(i<n)
        {
            if(st.empty()||arr[st.top()]<=arr[i])
            {
                st.push(i++);
            }
            else
            {
                int tp=st.top();
                st.pop();
                long long ar=arr[tp]*(st.empty()?i:(i-st.top()-1));
                max_ar=max(max_ar,ar);
            }
