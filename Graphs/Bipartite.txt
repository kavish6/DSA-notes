check if a graph is bipartitie or not (bipartite means if its vertices can be put in two sets which only have inter edges not intra) 


// we think of all vertices having one of 2 colors if adjacent vertices have same color the graph is not bipartite


 bool dfs(int i,vector<int> adj[],vector<bool>& vis,vector<int>& col,int par)
    {
        vis[i]=true;
        if(par==-1)
        col[i]=1;
        else
        {
            if(col[i]==0)
            {
                col[i]=-col[par];
            }
            else
            {
                if(col[i]==col[par])
                return true;
            }
        }
        for(auto it:adj[i])
        {
            par=i;
            if(!vis[it])
            {
                vis[it]=true;
                if(dfs(it,adj,vis,col,par))
                return true;
            }
            else
            {
                if(col[i]==col[it]) 
                return true;
            }
        }
        return false;
    }
	bool isBipartite(int V, vector<int>adj[]){
	    // Code here
	    vector<bool> vis(V,false);
	    vector<int> col(V,0);
	    int par=-1;
	    for(int i=0;i<V;i++)
	    {
	        if(!vis[i])
	        {
	        if(dfs(i,adj,vis,col,par))
	        return false;
	        }
	    }
	    return true;
	    
	}
